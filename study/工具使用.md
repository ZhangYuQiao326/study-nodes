```cpp
//.h
#ifndef DB_H
#define DB_H

#include <mysql/mysql.h>
#include <string>
using namespace std;

// 数据库操作类
class MySQL
{
public:
    // 初始化数据库连接
    MySQL();
    // 释放数据库连接资源
    ~MySQL();
    // 连接数据库
    bool connect();
    // 更新操作
    bool update(string sql);
    // 查询操作
    MYSQL_RES *query(string sql);
    // 获取连接
    MYSQL *getConnection();

private:
    MYSQL *_conn;
};

#endif
```

```cpp
// .cpp
#include "db.h"
#include <muduo/base/Logging.h>

// 数据库配置信息
static string server = "127.0.0.1";
static string user = "root";
static string password = "123456";
static string dbname = "chat";

// 初始化数据库连接
MySQL::MySQL()
{
    _conn = mysql_init(nullptr);
}

// 释放数据库连接资源
MySQL::~MySQL()
{
    if (_conn != nullptr)
        mysql_close(_conn);
}

// 连接数据库
bool MySQL::connect()
{
    MYSQL *p = mysql_real_connect(_conn, server.c_str(), user.c_str(),
                                  password.c_str(), dbname.c_str(), 3306, nullptr, 0);
    if (p != nullptr)
    {
        // C和C++代码默认的编码字符是ASCII，如果不设置，从MySQL上拉下来的中文显示？
        mysql_query(_conn, "set names gbk");
        LOG_INFO << "connect mysql success!";
    }
    else
    {
        LOG_INFO << "connect mysql fail!";
    }

    return p;
}

// 更新操作
bool MySQL::update(string sql)
{
    if (mysql_query(_conn, sql.c_str()))
    {
        LOG_INFO << __FILE__ << ":" << __LINE__ << ":"
                 << sql << "更新失败!";
        return false;
    }

    return true;
}

// 查询操作
MYSQL_RES *MySQL::query(string sql)
{
    if (mysql_query(_conn, sql.c_str()))
    {
        LOG_INFO << __FILE__ << ":" << __LINE__ << ":"
                 << sql << "查询失败!";
        return nullptr;
    }

    return mysql_use_result(_conn);
}

// 获取连接
MYSQL *MySQL::getConnection()
{
    return _conn;
}
```

# 五 shell脚本

Shell脚本语法包括了许多构造和概念，这些是编写有效Shell脚本的基础。下面是一些常见的Shell脚本语法和概念。

### 1. Shebang（#!）
脚本的第一行通常是Shebang（`#!`），后面跟着用于解释该脚本的解释器的路径。对于Bash脚本，它看起来像这样：
```bash
#!/bin/bash
```

### 2. 变量
在Shell脚本中，你可以定义变量来存储数据，如：
```bash
name="World"
echo "Hello, $name!"
```
变量前面不需要声明类型，直接使用即可。使用变量时，推荐用双引号包围变量，以避免空格和特殊字符造成的问题。

### 3. 条件语句
条件语句允许你根据条件执行不同的代码段。
```bash
if [ $name == "World" ]; then
  echo "Hello, World!"
else
  echo "Unknown"
fi
```

### 4. 循环
Shell脚本支持`for`循环、`while`循环和`until`循环。
- **For循环**：
  ```bash
  for i in {1..5}; do
    echo "Number $i"
  done
  ```
- **While循环**：
  ```bash
  count=1
  while [ $count -le 5 ]; do
    echo "Count $count"
    ((count++))
  done
  ```

### 5. 函数
函数让你可以将代码组织成可重复使用的单元。
```bash
greet() {
  echo "Hello, $1!"
}
greet World
```

### 6. 读取输入
使用`read`命令读取用户输入：
```bash
echo "What's your name?"
read name
echo "Hello, $name!"
```

### 7. 引用
- **单引号** `'` 用于字面量字符串，不进行变量扩展。
- **双引号** `"` 用于需要解析变量或执行命令的字符串。

### 8. 文件和目录操作
使用标准的Unix命令，如`cp`, `mv`, `mkdir`, `rm`等进行文件和目录操作。

### 9. 条件表达式
- **文件测试运算符**：例如，`-e` 检查文件是否存在，`-d` 检查目录是否存在。
- **字符串比较**：例如，`$str1 == $str2` 检查两个字符串是否相等。
- **算术比较**：例如，`-lt`, `-gt` 分别用于小于和大于比较。

### 10. 错误处理
使用`set -e`让脚本在遇到错误时停止执行，使用`trap`捕获信号和结束动作。

# 六 json

* 头文件

> 只有头文件，因此不需要链接库

```cpp
#include "json.hpp"
using json = nlohmann::json;

```

* 序列化

```cpp
json js;
// 添加数组
js["id"] = {1,2,3,4,5};
// 添加key-value
js["name"] = "zhang san";
// 添加对象
js["msg"]["zhang san"] = "hello world";
js["msg"]["liu shuo"] = "hello china";
// 上面等同于下面这句一次性添加数组对象
js["msg"] = {{"zhang san", "hello world"}, {"liu shuo", "hello china"}};
cout << js << endl;

```

* 容器序列化

```cpp
json js;
// 直接序列化一个vector容器
vector<int> vec;
vec.push_back(1);
vec.push_back(2);
vec.push_back(5);
js["list"] = vec;
// 直接序列化一个map容器
map<int, string> m;
m.insert({1, "黄山"});
m.insert({2, "华山"});
m.insert({3, "泰山"});
js["path"] = m;
cout<<js<<endl;
```

* 反序列化

```cpp
string jsonstr = js.dump();
cout<<"jsonstr:"<<jsonstr<<endl;

// 模拟从网络接收到json字符串，通过json::parse函数把json字符串专程json对象
json js2 = json::parse(jsonstr);
// 直接取key-value
string name = js2["name"];
cout << "name:" << name << endl;
// 直接反序列化vector容器
vector<int> v = js2["list"];
for(int val : v)
{
cout << val << " ";
}
cout << endl;
// 直接反序列化map容器
map<int, string> m2 = js2["path"];
for(auto p : m2)
{
cout << p.first << " " << p.second << endl;
}
cout << endl;
```
